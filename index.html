Simple to Use

Just drop in the components and customize with attributes

<!-- Include dependencies -->
<script src="https://cdn.tailwindcss.com"></script>
<script src="https://unpkg.com/alpinejs@3.x.x/dist/cdn.min.js" defer></script>

<!-- Include DRY2 complete bundle (all 21 components) -->
<script src="https://unpkg.com/dry2-web-components/dist/dry2.js"></script>

<!-- Use any component - they're all available! -->
<dry-button variant="primary" size="lg">Click me!</dry-button>
<avatar-component name="John Doe" size="lg"></avatar-component>
<carousel-component slides='[...]' autoplay></carousel-component>
<toast-component type="success" auto-show>Success!</toast-component>
